When the agent finishes a turn, display a compact summary of what happened: files changed, commands run, time elapsed, errors.

## What to implement

### Summary generation

Add a method to `TurnTracker`:
```rust
impl TurnTracker {
    pub fn summary(&self) -> Option<String> {
        if self.tool_calls_completed == 0 {
            return None;
        }
        
        let elapsed = self.elapsed();
        let time = format!("{}:{:02}", elapsed.as_secs() / 60, elapsed.as_secs() % 60);
        
        let mut parts = Vec::new();
        
        // Files modified
        if !self.files_written.is_empty() {
            let count = self.files_written.len();
            parts.push(format!("{count} file{} modified", if count > 1 { "s" } else { "" }));
        }
        
        // Files read (only if no writes — otherwise it's noise)
        if self.files_written.is_empty() && !self.files_read.is_empty() {
            let count = self.files_read.len();
            parts.push(format!("{count} file{} read", if count > 1 { "s" } else { "" }));
        }
        
        // Commands run
        if !self.commands_run.is_empty() {
            let count = self.commands_run.len();
            parts.push(format!("{count} command{}", if count > 1 { "s" } else { "" }));
        }
        
        // Errors
        if self.tool_errors > 0 {
            parts.push(format!("{} error{}", self.tool_errors, if self.tool_errors > 1 { "s" } else { "" }));
        }
        
        let detail = parts.join(", ");
        let mut summary = format!("Done in {time} — {detail}");
        
        // List modified files
        if !self.files_written.is_empty() {
            summary.push_str("\n  Files:");
            for f in &self.files_written {
                // Shorten path: strip cwd prefix if present
                summary.push_str(&format!("\n    • {f}"));
            }
        }
        
        Some(summary)
    }
}
```
